FROM  inventory
JOIN images
ON inventory.itemID=images.itemID
JOIN colors
ON inventory.colorID=colors.colorID
JOIN parts
ON inventory.itemID=parts.PartID
JOIN itemtypes
ON inventory.itemTypeID=images.itemTypeID
WHERE 1
AND inventory.SetID='045-1'

SELECT *
FROM inventory
JOIN itemtypes
ON inventory.itemTypeID=itemtypes.itemTypeID
JOIN images
ON itemtypes.itemTypeID=images.itemTypeID
WHERE 1
ORDER BY itemtypes.itemTypeID DESC
LIMIT 30
